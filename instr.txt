build and install deb:
rm -rf dist
npm run dist:linux
sudo apt install "./dist/CanSat_Terminal-1.0.0-linux-amd64.deb"

uninstall:
sudo apt remove cansat-terminal-electron

add user to dialout (access to serial ports):
sudo usermod -aG dialout $USER
